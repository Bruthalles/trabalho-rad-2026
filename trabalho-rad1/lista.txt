import tkinter

def _test():
    root = tkinter.Tk()
    text = "This is Tcl/Tk %s" % root.globalgetvar('tk_patchLevel')
    text += "\nThis should be a cedilla: \xe7"
    
    label = tkinter.Label(root, text=text)
    label.pack()
    
    test = tkinter.Button(root, text="Click me!")
    
    test['command'] = lambda: test.configure(text="[%s]" % test['text'])
    test.pack()
    
    quit = tkinter.Button(root, text="QUIT", command=root.destroy)
    quit.pack()

    root.iconify() #Minimiza a tela
    root.update()
    root.deiconify() #Maximiza a tela
    root.mainloop()  #loop principal, impede o código de seguir e permite capturar inputs
